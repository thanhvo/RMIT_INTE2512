package sample;

import javafx.application.Application;
import javafx.fxml.FXMLLoader;
import javafx.scene.Parent;
import javafx.scene.Scene;
import javafx.scene.input.KeyCombination;
import javafx.scene.media.Media;
import javafx.scene.media.MediaPlayer;
import javafx.stage.*;

import java.io.File;
import java.io.FileInputStream;
import java.io.InputStream;
import java.util.Properties;


public class Main extends Application {


    private Properties properties = new Properties();
    public static MediaPlayer mediaPlayer;



    // primaryStage is the main stage of the game
    @Override
    public void start(Stage primaryStage) throws Exception{

        String musicFile = "src/Sounds/Closer.mp3";     // Play music from directory File

        Media sound = new Media(new File(musicFile).toURI().toString());
        mediaPlayer = new MediaPlayer(sound);
        mediaPlayer.play();

        Parent root = FXMLLoader.load(getClass().getResource("views/MainMenu.fxml"));

        double width = 1280,height =720;
        boolean fullScreen;

        File f = new File("config.properties");
        if(f.exists()) {
            InputStream input = new FileInputStream("config.properties");
            properties.load(input);

            width = Double.parseDouble(properties.getProperty("width"));
            height = Double.parseDouble(properties.getProperty("height"));
            fullScreen = Boolean.parseBoolean(properties.getProperty("fullScreen"));
            primaryStage.setFullScreen(fullScreen);
            primaryStage.setFullScreenExitHint("");
            primaryStage.setFullScreenExitKeyCombination(KeyCombination.NO_MATCH);
        }

        primaryStage.setTitle("Memory Game");
        Scene scene = new Scene(root,width,height);
        primaryStage.setScene(scene);
        primaryStage.setResizable(false);
        
primaryStage.show();

    }


    public static void main(String[] args) {
        launch(args);
    }
}

package sample.controller;

import javafx.application.Platform;
import javafx.fxml.FXML;
import javafx.scene.Parent;

import javafx.scene.control.Button;
import javafx.scene.control.Label;
import javafx.scene.media.Media;
import javafx.scene.media.MediaPlayer;
import javafx.stage.Stage;
import javafx.util.Duration;

import java.io.*;
import java.util.Locale;
import java.util.Properties;
import java.util.ResourceBundle;

public class ExitController {

    @FXML
    private Button yes,no;

    @FXML
    private Label labelExit;

    private Properties properties = new Properties();
    private MediaPlayer mediaPlayer;

    // Manufacturer of class
    public ExitController(){
        Media buttonSound = new Media(new File("src/Sounds/buttonSound.wav").toURI().toString());
        mediaPlayer = new MediaPlayer(buttonSound);
    }

    // Make initialize throws Exception
    @FXML
    private void initialize() throws IOException{       // [1]
        File f = new File("config.properties");

        if(f.exists()) {
            InputStream input = new FileInputStream("config.properties");
            properties.load(input);

            String lang = properties.getProperty("flag");
            loadLang(lang);
        }
    }

    // The event handler of  YES button.
    @FXML
    private void yesClicked(){          //[2]
        mediaPlayer.seek(Duration.ZERO);
        mediaPlayer.setVolume(0.3f);
        mediaPlayer.play();
        Platform.exit();
    }
    //The event handler of the NO button.
    @FXML
    private void noClicked() {   //[2]
        mediaPlayer.seek(Duration.ZERO);
        mediaPlayer.setVolume(0.3f);
        mediaPlayer.play();
        Stage dialog = (Stage) no.getScene().getWindow();

        Parent root = dialog.getOwner().getScene().getRoot();
        root.setEffect(null);

        dialog.close();
    }


     //Loads the language of the game.
    private void loadLang(String lang) {       // [3]
        Locale locale = new Locale(lang);
        ResourceBundle bundle = 
ResourceBundle.getBundle("sample.lang", locale);

        labelExit.setText(bundle.getString("exitDialog"));
        yes.setText(bundle.getString("yes"));
        no.setText(bundle.getString("no"));
    }

}

    public GamePlayController() {
        gameMode = new GameMode();

        cards = new ArrayList<>();

    // Loads values from score and config files.
    @FXML
    private void initialize() throws IOException {    // [1]

    // Specifies the type of game and the subject of the cards depending on what the user has chosen in the settings.
    public void setMode(GameMode gameMode, Image theme) throws IOException {
        this.gameMode = gameMode;
        this.theme = theme;

        
gameMode.CreateMode();
        gameMode.OverallSceneSettings();

    }
    // Creates the board and cards and shuffles them. Then invokes the ImageView event handler to get the player started playing.
    public void gameStart() {
        createImageViews(gridCard, imageViews);
        createImages(cards);
        shuffleCards(imageViews);
        setImages(imageViews, cards);

        player();
    }

    // He is the event handler of the cards.
    public void player() {
        for (int i = 0; i < imageViews.size(); i++) {
            final ImageView imageView = imageViews.get(i);
            final Card card = cards.get(i);
            imageView.setPreserveRatio(false);
            imageViews.get(i).setOnMouseClicked(event -> {
                try {
                    
clickEvent(imageView, card);
                } catch (IOException e) {
                    e.printStackTrace();
                }
            });
        }
    }


    //If an ImageView is pressed then this method is called. It returns with the help of a ScaleTransition to turn the card that the user has selected.
    //second or third GameMode and compares them, while saving some cards and ImageView to ArrayList.
    public void clickEvent(ImageView imageView, Card card) throws IOException {
        cardsMatch = false;
        clicks++;
        imageView.setDisable(true);
        ScaleTransition scaleTransition = new ScaleTransition(Duration.seconds(0.4), imageView);
        scaleTransition.setFromX(1);
        scaleTransition.setToX(-1);
        scaleTransition.play();
        scaleTransition.setOnFinished(event -> {
            imageView.setScaleX(1);
            imageView.setImage(card.getValue());
        });

        if (clicks == 1) {
            id1 = card.getId();
            imageView1 = imageView;
            card1 = card;

        }
        if (clicks == 2) {
            id2 = card.getId();
            imageView2 = imageView;
            card2 = card;

                GameSingleton.getInstance().getScore().updateMoves();
                if (gameMode.getGlobalMode().equals("SingleMode"))
                    MoveCountLabel.setText(word1 + GameSingleton.getInstance().getScore().getMoves());
                disableAll();
                if (id1 == id2) {
                    GameSingleton.getInstance().getScore().updateFoundCards();
                    if (gameMode.getGlobalMode().equals("SingleMode"))
                        pairFoundLabel.setText(word2 + GameSingleton.getInstance().getScore().getFoundCards());
                    cardsMatch = true;
                    foundCards.add(imageView1);
                    foundCards.add(imageView2);

                    Timeline timeline = new Timeline(new KeyFrame(Duration.seconds(0.6), event -> {
                        imageView1.setDisable(true);
                        imageView2.setDisable(true);
                        imageView1.setOpacity(0.6);
                        imageView2.setOpacity(0.6);
                        if (gameMode.getGlobalMode().equals("SingleMode"))
                            enableAll();
                    }));
                    timeline.play();
                } else {
                    System.out.println(gameMode.getMode());
                    if(gameMode.getMode() == 1){
                        displaySecond = 3;
                    }
                    if(gameMode.getMode() == 2){
                        displaySecond = 2;
                    }
                    if(gameMode.getMode() == 3){
                        displaySecond = 1;
                    }


                    Timeline timeline = new Timeline(new KeyFrame(Duration.seconds(displaySecond), event -> {
                        imageView1.setImage(card1.getBackground());
                        imageView2.setImage(card2.getBackground());
                        if (gameMode.getGlobalMode().equals("SingleMode")) {
                            imageView1.setDisable(false);
                            imageView2.setDisable(false);
                            enableAll();
                        }
                    }));
                    timeline.play();
                }
                if (foundCards.size() == gameMode.getSize() && gameMode.getGlobalMode().equals("SingleMode")) {
                    Timeline timeline = new Timeline(new KeyFrame(Duration.seconds(1.2), event -> eraseCards(gridCard)));
                    timeline.play();
                    endGame = true;
                    Alert a = new Alert(Alert.AlertType.CONFIRMATION);
                    currentScore += 120 -(120 -(counter / 100));

                    GameSingleton.getInstance().getScore().setCurrentScore(GameSingleton.getInstance().getScore().getCurrentScore() + currentScore);
                    a.setContentText("Your Score is: " + GameSingleton.getInstance().getScore().getCurrentScore() + "\n"   + "Press Ok to PLAY AGAIN or Cancel to EXIT to the MAIN MENU" );

                    Optional<ButtonType> result = a.showAndWait();
                    if(result.isPresent() && result.get() == ButtonType.OK){
                        Parent root = FXMLLoader.load(getClass().getResource("../views/ModeSettings.fxml"));
                        Stage stage = (Stage) gamePlayPane.getScene().getWindow();
                        stage.getScene().setRoot(root);
                    }else if (result.isPresent() && result.get() == ButtonType.CANCEL){
                        Parent root = FXMLLoader.load(getClass().getResource("../views/MainMenu.fxml"));
                        Stage stage = (Stage) gamePlayPane.getScene().getWindow();
                        stage.getScene().setRoot(root);

                    }
                }
                clicks = 0;
        }

    }

    // Deletes the cards from GridPane when the game is over.
    public void eraseCards(GridPane grid) 
{
        for (int i = 0; i < imageViews.size(); i++) {
            grid.getChildren().remove(imageViews.get(i));
        }
    }

    // Activates all cards except those found so that the user can touch them.
    public void enableAll() {
        for (int i = 0; i < imageViews.size(); i++) {
            imageViews.get(i).setDisable(false);
        }

        for (int i = 0; i < foundCards.size(); i++) {
            foundCards.get(i).setDisable(true);
        }
    }

    // Disables all cards.
    public void disableAll() {
        for (int i = 0; i < imageViews.size(); i++) {
            imageViews.get(i).setDisable(true);
        }
    }




    //Takes the GridPane created in the FXML file, creates rows and columns accordingly,
    // leaving the appropriate spaces and setting the appropriate dimensions.
    public void createImageViews(GridPane grid, ArrayList<ImageView> imageViews) {

        grid.setHgap(10);
        grid.setVgap(10);
        // Creates ImageView objects and puts them in ArrayList.
        for (int i = 0; i < gameMode.getRows(); i++) {
            RowConstraints row = new RowConstraints(gameMode.getImHeight());
            row.setMinHeight(Double.MIN_VALUE);
            row.setVgrow(Priority.ALWAYS);
            grid.getRowConstraints().add(row);
        }
        for (int i = 0; i < gameMode.getColumns(); i++) {
            ColumnConstraints column = new ColumnConstraints(gameMode.getImWidth());
            column.setMinWidth(Double.MIN_VALUE);
            column.setHgrow(Priority.ALWAYS);
            grid.getColumnConstraints().add(column);
        }

        for (int i = 0; i < gameMode.getRows(); i++) {
            for (int j = 0; j < gameMode.getColumns(); j++) {
                ImageView imageView = new ImageView();
                imageView.setPreserveRatio(true);
                imageView.setFitWidth(gameMode.getImWidth());
                imageView.setFitHeight(
gameMode.getImHeight());
                grid.add(imageView, j, i);
                imageViews.add(imageView);
            }
        }

    }

    // Takes pictures from the Images folder, creates the cards and saves them to ArrayList.
    public void createImages(ArrayList<Card> cards) {
        int times = 0;
        int j = 0;
        for (int i = 1; i <= gameMode.getSize(); i++) {
            if (i % gameMode.getSelectCards() == 1) {
                times++;
                j++;
            }
            Image cardImage = new Image("Images/Cards/" + j + ".png");
            Card image2 = new Card(cardImage, theme, times);
            cards.add(image2);
        }
    }


    public void setImages(ArrayList<ImageView> imageViews, ArrayList<Card> cards) {
        for (int i = 0; i < imageViews.size(); i++) {
            imageViews.get(i).setImage(cards.get(i).getBackground());
        }
    }


    public void shuffleCards(ArrayList<ImageView> imageViews) {
        Collections.shuffle(imageViews);
    }


    private void loadLang(String lang) {
        Locale locale = new Locale(lang);
        ResourceBundle bundle = ResourceBundle.getBundle("sample.lang", locale);

        MoveCountLabel.setText(bundle.getString("moves"));
        pairFoundLabel.setText(bundle.getString("foundPairs"));

    }

}

    // set the time for each game play for 2 minutes
    int counter = 12000;
    TextField timeCountDown;
    Timer timer = new Timer();

    
public TimerThread( TextField timeCountDown) {
        this.timeCountDown = timeCountDown;
    }
        @Override
        public void run() {
            // TODO Auto-generated method stub
            // calculate the minutes, seconds, and miliseconds
            int minutes= this.counter / 6000;
            int seconds = (this.counter % 6000) / 100;
            int mili = (this.counter % 600) % 100;
            // display the counter in type of string
            System.out.println(minutes + " : " + seconds + " : " +mili);
            counter--;
            System.out.println("TimerTask executing counter is: " + this.counter);
            timeCountDown.setText(String.valueOf(this.counter));
            if (this.counter == -1) { // stop the counter when the counter is equal to 0
                this.cancel();
                //                    isIt = true;
            }
            timeCountDown.setDisable(true);
            this.counter--;
        }
}
