    void setCardImage(int cardNum, Button bt) throws IOException {
        try {
            if (cardRunnning < 2){ //two buttons can be pressed
                cardRunnning++;
                if (cardNum == prevCardIndex){
                    return;
                }
                ImageView changeCardPic = new ImageView(playerCards[cardNum - 1]);
                changeCardPic.setPreserveRatio(true);
                changeCardPic.setSmooth(true);
                changeCardPic.setFitWidth(100);
                bt.setGraphic(changeCardPic);
                if (prevCard.equals(playerCards[cardNum - 1])){
                    matchedGameCount++;
                    if (matchedGameCount == 10){ //count how many pairs were matched and if 10 pairs are matched, show winGame
                        winGame();
                        return;
                    }
                    cardRunnning--; //if matched, count one down
                    prevCard = playerCards[cardNum - 1]; //set current image as prevCard
                    flag = 1;
                    return;
                }
                prevCard = playerCards[cardNum - 1];
                prevCardIndex = cardNum;
                //turn back
                setTimer(bt);
            }
        } catch(NullPointerException e) {
            System.out.println("error: " + e.getMessage());
        }
    }

    void setTimer(Button bt){
        Timer timer = new Timer();
        int levelTime = 0;
        //setting time by level
        if (MainPanelController.levelIn == 1){
            levelTime = 3000;
        } else if (MainPanelController.levelIn == 2){
            levelTime = 2000;
        } else if (MainPanelController.levelIn == 3){
            levelTime = 1000;
        }
        timer.schedule(new TimerTask() {
            @Override
            public void run() {
                if (flag == 1){
                    cardRunnning--;
                    flag = 0;
                    return;
                }
                Platform.runLater(() -> { //turn back after set time
                    turnCardBack(bt);
                    cardRunnning--;
                });
            }
        }, levelTime);
    }

    void turnCardBack(Button bt){
        try {
            ImageView cardBack = new ImageView("file:src/images/soccerBall.png");
            cardBack.setPreserveRatio(true);
            cardBack.setSmooth(true);
            cardBack.setFitWidth(100);
            bt.setGraphic(cardBack);
        } catch (NullPointerException e){
            System.out.println("Error: " + e.getMessage());
        }
    }

    void timeBar(){ //graphical time count
        try {
            timer.schedule(new TimerTask() {
                @Override
                public void run() {
                    if (flag == 2){ //if win game before time done, don't count down anymore
                        counter = 120001;
                    } else if (counter <= 119999){
                        Platform.runLater(() -> {
                            timeBar();
                            stringTime();
                            remainTime.setWidth(remainTime.getWidth() - 0.00333);
                            counter++;
                        });
                    } else if (counter == 120000){ //if not finished until count down, show lose game panel
                        try {
                            stringTime();
                            loseGame();
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                }
            }, 1);
        } catch (NullPointerException e){
            System.out.println("error: " + e.getMessage());
        }
    }

    void stringTime() { //timer
        try{
            timeText.setText(String.format("%02d:%02d:%02d", counter / 1000 / 60, (counter / 1000) % 60, (counter % 1000) / 10));
            scoreCount();
        } catch (NullPointerException e){
            System.out.println("error ");
        }
    }

    void winGame() throws IOException { //win game panel

     void scoreCount() { //counting score
        try{
            int scoreTemp = 0;
            int temp;
            int scoreDigitCount = 0;
            String scoreStringTemp = "";
            String scoreString = "";
            if (counter % 1000 == 0){ //point decrease at every second
                score -= 1;
                if (score < 1){
                    return;
                }
                //convert int score to sting to show up in panel because setText only accept String type
                scoreTemp = score;
                while (scoreTemp != 0){
                    temp = scoreTemp % 10;
                    scoreTemp -= temp;
                    scoreTemp /= 10;
                    scoreDigitCount++;
                }
                scoreTemp = score;
                for (int i = 0; i < scoreDigitCount; i++) {
                    temp = scoreTemp % 10;
                    scoreTemp -= temp;
                    scoreTemp /= 10;
                    temp += 48;
                    scoreStringTemp += (char)temp;
                }
                for (int i = scoreDigitCount - 1; i >= 0; i--) {
                    scoreString += scoreStringTemp.charAt(i);
                }
                scoreText.setText(scoreString); //change score in panel
            }
        }catch (NullPointerException e){
            System.out.println("error: " + e.getMessage());
        }
     }

public class MainPanelController implements Initializable {

    @FXML
    private ComboBox<Integer> levelInput;

    @FXML
    private Button startButton;

    @FXML
    void levelInputEvent(ActionEvent event) {
        levelIn = levelInput.getValue();
    }

    @FXML
    void starts(ActionEvent event) throws IOException {
        try {
            Parent root = FXMLLoader.load(getClass().getResource("GamePanel.fxml"));
            Scene scene = new Scene(root);
            Stage stage = new Stage(StageStyle.UNDECORATED);
            stage.setScene(scene);
            stage.show();
            cardShuffle();
        } catch (NullPointerException e){
            System.out.println("Error: " + e.getMessage());
        }
        //GamePanelController.paneSetting();
    }

    public static int levelIn;
    private void cardShuffle() {
        try {
            Random rnd = new Random();
            String temp;
            ArrayList<Integer> num = new ArrayList<>();
            for (int k = 0; k < 20; k++) {
                int l = rnd.nextInt(20);
                if (num.contains(l)){
                    while(num.contains(l)){
                        l = rnd.nextInt(20);
                    }
                }
                num.add(l);
                temp = GamePanelController.playerCards[k];
                GamePanelController.playerCards[k] = GamePanelController.playerCards[l];
                GamePanelController.playerCards[l] = temp;
            }
        } catch (NullPointerException e ){
            System.out.println("Error: " + e.getMessage());
        }
    }
    @Override
    public void initialize(URL arg0, ResourceBundle arg1) {
        ObservableList<Integer> levelList = FXCollections.observableArrayList();
        levelInput.setItems(levelList);
        for (int i = 0; i < 4; i++) {
            if (i != 0){
                levelList.add(i);
            }
        }
    }
}
